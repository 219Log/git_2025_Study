	pageEncoding="UTF-8"%>

<%@ taglib prefix="fn" uri="http://java.sun.com/jsp/jstl/functions"%>
<%@ taglib prefix="form" uri="http://www.springframework.org/tags/form"%>

<script type="text/javascript">
	var updatedQuestList = []; // 수정된 응답항목 목록
	var deletedQuestList = []; // 삭제된 응답항목 목록

	$(function() {
		// 신청자이름/ 이메일은 필수. 변경불가.
		$("[id*=E01A03A01").attr("disabled", "disabled");
		$("[id*=E01A03A02").attr("disabled", "disabled");
		// 공고정보설정 버튼 클릭
		$('#btnPbanc').click(function(e) {
			e.preventDefault();
			var frm = document.regForm;
			frm.action = "<c:url value='${referer}' />";
			frm.submit();
		});

		// 텍스트 입력 시 자동 높이 조절
		function autoResize(e) {
			$(e).css('height', 'auto');
			$(e).css('height', e.scrollHeight + 'px');
		}

		// 기존 텍스트에리어에 대해 초기 높이 설정
		$('.question-title').each(function() {
			autoResize(this);
		});

		// 입력 시 높이 자동 조절
		$(document).on('input', '.question-title', function() {
			autoResize(this);
		});

		// 본인인증 체크박스 클릭 시 자동으로 스위치 ON
		$("#useYn_E01A03A03").on('change', function() {
			if ($(this).is(':checked')) {
				$("#esntlYn_E01A03A03").prop('checked', true);
			}
		});

		// 필수 스위치가 변경될 때 본인인증 체크박스 상태 확인
		$("#esntlYn_E01A03A03").on('change', function() {
			if ($("#useYn_E01A03A03").is(':checked')) {
				$("#useYn_E01A03A03").prop('checked', false);
			}
		});

		// 저장 버튼 클릭 이벤트
		$('#btnSave').click(function(e) {
			e.preventDefault();
			saveData("save"); // 저장 후 목록으로 이동
		});

		// 미리보기 버튼 클릭 이벤트
		$('#btnPrev').click(function(e) {
			e.preventDefault();
			saveData("preview"); // 저장 후 미리보기로 이동
		});

		// 승인 절차 신청 버튼 클릭 이벤트
		$("#btnAprvAply").click(
				function(e) {
					e.preventDefault();
					if ("Y" != $("#picYn").val()) {
						alert('해당 공고의 담당자만 수정할 수 있습니다.', function() {
						});
						return;
					}

					// 개인정보 수집 및 이용 내용
					var prvcPlcyCn = $('#textarea1').val();

					// 개인정보 수집 및 이용 내용 확인 체크
					var prvcClctAgreYn = $('#checkbox1').is(':checked');

					// 필수 체크
					if (!prvcPlcyCn) {
						alert('개인정보 수집 및 이용 내용을 입력해주세요.', function() {
							$('#textarea1').focus();
						});
						return;
					}

					if (!prvcClctAgreYn) {
						alert('개인정보 수집 및 이용 내용 확인이 필요합니다.', function() {
							$('#checkbox1').focus();
						});
						return;
					}

					let emptyInput = null;

					// file-extn-lmt 클래스를 가진 입력 필드를 제외하고 검사
					$("input[type='text']:not(.file-extn-lmt), textarea").each(
							function() {
								if ($(this).val().trim() === "") {
									emptyInput = $(this); // 첫 번째 빈 입력 필드 저장
									return false; // 반복문 중단
								}
							});

					if (emptyInput) {
						alert('빈 값이 있습니다. 값을 입력해주세요.', function() {
							emptyInput.focus(); // 빈 입력 필드에 포커스
						});
						return;
					}

					saveData("approve");
				});

		// 목록 이동 함수
		function fn_goList() {
			var frm = document.regForm;
			frm.action = '<c:url value="/survey/onlnRcpt/onlnRcptList.do" />';
			frm.submit();
		}

		// 데이터 저장 함수
		function saveData(actionType) {
			// 현재 공고번호 가져오기
			var pbancNo = '${resultVO.pbancNo}';
			if (!pbancNo) {
				alert('공고번호가 없습니다.');
				return;
			}

			// 개인정보 동의 활용 기간 체크 (true: 3개월, false: 6개월)
			var prvcStrgTermCd = $('input[name="prvcStrgTermCd"]').is(
					':checked');

			// 개인정보 수집 및 이용 내용
			var prvcPlcyCn = $('#textarea1').val();

			// 개인정보 수집 및 이용 내용 확인 체크
			var prvcClctAgreYn = $('#checkbox1').is(':checked');

			// 필수 체크
			if (!prvcPlcyCn) {
				alert('개인정보 수집 및 이용 내용을 입력해주세요.', function() {
					$('#textarea1').focus();
				});
				return;
			}

			if (!prvcClctAgreYn) {
				alert('개인정보 수집 및 이용 내용 확인이 필요합니다.', function() {
					$('#checkbox1').focus();
				});
				return;
			}

			// 미입력 항목값 체크 변수
			var hasEmpty2 = false;

			// 1. 개인정보 항목설정 데이터 수집
			var rspnsArtclList = [];
			$("[id^=useYn").each(function() {
				var id = $(this).attr("id").replace('useYn_', '');
				rspnsArtclList.push({
					pbancNo : pbancNo,
					artclCd : id,
					useYn : $('#useYn_' + id).is(':checked') ? 'Y' : 'N',
					esntlYn : $('#esntlYn_' + id).is(':checked') ? 'Y' : 'N'
				});
			});

			// 2. 질문 유형 데이터 수집
			var updatedQuestList = [];
			var hasEmptyQuestion = false; // 질문 미입력 플래그
			var hasEmptyOption = false; // 선택항목 미입력 플래그
			var $emptyQuestionBox = null; // 미입력 질문 박스

			$('.re_wr_box')
					.each(
							function(index) {
								var $box = $(this);
								var qiteid = $box.attr("id");

								var type = $box.find('.choice_sel_active')
										.attr('data-type');
								var qitemId = $box.attr('id');
								var qitemCn = $box.find('.question-title')
										.val();

								// 질문 내용 검사
								if (qitemCn == '') {
									hasEmptyQuestion = true;
									if (!$emptyQuestionBox) {
										$emptyQuestionBox = $box; // 첫 번째 미입력 질문 저장
									}
								}

								var esntlYn = $box.find('.switch-input').is(
										':checked') ? 'Y' : 'N';
								var maxLen = $box.find('.text-limit').val();
								var maxChcCnt = $box.find('.checkbox-limit')
										.val();

								var questionData = {
									qiteid : qiteid,
									pbancNo : pbancNo,
									qitemId : qitemId,
									type : type,
									qitemCn : qitemCn,
									esntlYn : esntlYn,
									maxLen : maxLen,
									maxChcCnt : maxChcCnt,
									sortSn : index + 1
								};

								if ([ 'objective', 'checkbox', 'dropDown' ]
										.includes(type)) {
									var options = [];
									$box.find('.option-text').each(
											function(index) {
												var $opt = $(this);
												var anschcId = $opt.data('id');
												var anschcCn = $opt.val();

												if (anschcCn == '') {
													hasEmptyOption = true; // 선택항목 미입력 플래그
												}

												options.push({
													anschcId : anschcId,
													anschcCn : anschcCn,
													sortSn : index + 1
												});

											});
									questionData.options = options;
								} else if (type === 'file') {
									var fileExtnLmt = $box.find(
											'.file-extn-lmt').val();
									// 파일 확장자 검사 제외
									questionData.fileExtnLmt = fileExtnLmt;
								}

								updatedQuestList.push(questionData);
							});

			// 검사 결과에 따른 처리
			if (hasEmptyQuestion) {
				alert('질문 내용을 입력해주세요.');
				if ($emptyQuestionBox) {
					$emptyQuestionBox.find('.question-title').focus();
				}
				return;
			}

			if (hasEmptyOption) {
				alert('객관식/체크박스 문항의 선택지를 입력해주세요.');
				return;
			}

			// 서버 전송용 데이터 객체 구성
			var requestData = {
				pbancNo : pbancNo,
				prvcStrgTermCd : prvcStrgTermCd,
				prvcPlcyCn : prvcPlcyCn,
				prvcClctAgreYn : prvcClctAgreYn,
				rspnsArtclList : rspnsArtclList,
				updatedQuestList : updatedQuestList,
				deletedQuestList : deletedQuestList,
				actionType : actionType, // // 'preview', 'save', 'approve' 중 하나
				hasEmpty2 : hasEmpty2
			};

			// AJAX를 통한 서버 저장 요청
			$.ajax({
				url : '<c:url value="/survey/onlnRcpt/addRspnsArtclStngSave.do"/>',
				type : 'POST',
				data : JSON.stringify(requestData),
				contentType : 'application/json',
				dataType : 'json',
				success : function(response) {
					if (response.success) {
						// 미리보기 팝업 열기
						if (actionType === "preview") {
							fn_popup('<c:url value="/survey/onlnRcpt/rspnsArtclPrvw.do?pbancNo=' + pbancNo + '"/>');
						// 승인 절차 신청
						} else if (actionType === "approve") {
							confirm(
									'정상적으로 저장되었습니다.\r \n 개인정보 관리자에게 메일을 발송하여 공고 승인 절차를 진행하시겠습니까?',
									function(result) {
										if (result) {
											const url = '<c:url value="/survey/onlnRcpt/updateAgreeInfo.do"/>';
											const params = $("#regForm")
													.serializeArray();
											gfn_ajaxCall(
												url,
												params,
												function(data) {
													alert(
														data.message, function(result) {
																if (result) {
																	fn_goList(); // 목록 으로 이동
																}
															}
													);
											});
										}
									});
						// 임시저장
						} else {
							alert(response.message, function(result) {
								// 상세페이지(수정불가)로 이동하는걸로 바꾸기
								var frm = document.regForm;
								frm.action = '<c:url value="/survey/onlnRcpt/getPbancInfoStng.do" />';
								frm.submit();
							});
						}
					} else {
						alert(response.message || '저장 중 오류가 발생했습니다.');
					}
				},
				error : function(xhr, status, error) {
					console.error('Error:', error);
					alert('저장 중 오류가 발생했습니다. 관리자에게 문의하세요.');
				}
			});
		}

		// 기존 박스들에 대한 이벤트 핸들러 초기화
		fn_initializeExistingBoxes();

		var $container = $('.response_writing_area');

		// 선택항목 추가 (객관식/체크박스)
		$container
				.on(
						'click',
						'.add-option',
						function(e) {
							var $box = $(this).closest('.re_wr_box');
							var $optionList = $box.find('ul');
							var optionCount = $optionList.children('li').length + 1;
							var optionNm = '';
							if ($box.find('.choice_sel_active').attr(
									'data-type') === 'checkbox') {
								optionNm = '체크항목';
							} else {
								optionNm = '선택항목';
							}

							var template = '<li>'
									+ '<input type="text" class="option-text" data-id="" placeholder="' + optionNm + optionCount + '" maxlength="80">'
									+ '<button type="button" class="delete-option"><i class="xi-trash-o"></i></button>'
									+ '</li>';

							$optionList.append(template);

							// 체크박스인 경우 선택 가능 개수 업데이트
							if ($box.find('.choice_sel_active').attr(
									'data-type') === 'checkbox') {
								var $select = $box.find('.checkbox-limit');
								fn_generateCheckboxLimitOptions($select[0],
										optionCount);
							}
						});

		// 질문 삭제
		$container.on('click', '.delete-box', function(e) {
			var $box = $(this).closest('.re_wr_box');
			var qitemId = $box.find('.choice_sel_active').data('id');

			if (qitemId) {
				var deletedQuest = {
					qitemId : qitemId
				};
				deletedQuestList.push(deletedQuest);
			}
			$box.remove();
		});

		// 선택항목 삭제
		$container
				.on('click', '.delete-option',
						function(e) {
							var $box = $(this).closest('.re_wr_box');
							var $li = $(this).closest('li');
							var anschcId = $li.find('.option-text').data('id');
							var qitemId = $box.find('.choice_sel_active').data(
									'id');

							if (anschcId) {
								var deletedOptions = {
									qitemId : qitemId,
									anschcId : anschcId
								};
								deletedQuestList.push(deletedOptions);
							}

							$li.remove();

							// 체크박스인 경우 선택 가능 개수 업데이트
							if ($box.find('.choice_sel_active').attr(
									'data-type') === 'checkbox') {
								var $select = $box.find('.checkbox-limit');
								var remainingOptions = $box
										.find('.option-text').length;
								fn_generateCheckboxLimitOptions($select[0],
										remainingOptions);
							}
						});

		// 질문 유형 선택 토글
		$container.on('click', '.choice_sel_active',
				function(e) {
					var $list = $(this).closest('.choice_sel').find(
							'.choice_sel_list');
					$('.choice_sel_list').not($list).hide();
					$list.toggle();
				});

		// 질문 유형 변경
		$container
				.on(
						'click',
						'.choice_sel_list button',
						function() {
							var $box = $(this).closest('.re_wr_box');
							var type = $(this).attr('data-type');

							// 저장되어 있는 응답항목일 경우 기존 응답항목 삭제
							if ($box.attr('id')) {
								// 1. 해당박스의 id를 deletedQuestList 에 넣기
								var qitemId = $box.find('.choice_sel_active')
										.data('id');

								if (qitemId) {
									var deletedQuest = {
										qitemId : qitemId
									}
									deletedQuestList.push(deletedQuest);
								}

								// 2. 해당박스의 id와 셀렉트박스의 data-id를 빈값으로 변경
								$box.removeAttr('id');
								$box.find('.choice_sel_active').removeAttr(
										'data-id');
							}

							// 질문 유형에 따른 UI 업데이트
							var $active = $box.find('.choice_sel_active');
							$active.html($(this).html())
									.attr('data-type', type);
							$box.find('.choice_sel_list').hide();

							// 질문 유형에 따른 입력 영역 변경
							var $con = $box.find('.choice_text_con');
							switch (type) {
							case 'subjective':
							case 'long':
								$con
										.html('<ul><li>'
												+ '<label>최대 글자 수: </label>'
												+ '<select class="text-limit w20" name="text_limit"></select>'
												+ '</li></ul>');
								fn_generateTextLimitOptions($con
										.find('.text-limit')[0]);
								break;
							case 'objective':
							case 'checkbox':
								var template = '<div class="choice_text_'
										+ (type === 'objective' ? 'radio'
												: 'check')
										+ '">'
										+ '<ul><li>'
										+ '<input type="text" class="option-text" data-id="" placeholder="'
										+ (type === 'objective' ? '선택항목1'
												: '체크항목1')
										+ '" maxlength="80">'
										+ '<button type="button" class="delete-option"><i class="xi-trash-o"></i></button>'
										+ '</li></ul></div>'
										+ '<div class="choice_text_btn">'
										+ '<button type="button" class="btn_base bg_gray add-option">선택항목 추가</button>'
										+ '</div>';

								if (type === 'checkbox') {
									template += '<div class="checkbox-limit-wrap mt10">'
											+ '<label>체크 가능 개수: </label>'
											+ '<select class="checkbox-limit w20" name="checkbox_limit"></select>'
											+ '</div>';
								}
								$con.html(template);

								// 체크박스인 경우 선택 가능 개수 드롭다운 초기화
								if (type === 'checkbox') {
									var optionCount = $con.find('.option-text').length;
									fn_generateCheckboxLimitOptions($con
											.find('.checkbox-limit')[0],
											optionCount);
								}
								break;
							case 'file':
								$con
										.html('<ul><li>'
												+ '<input type="text" class="file-extn-lmt" placeholder="(미입력시 전체) 업로드 가능한 확장자를 입력해주세요. ex) ppt, zip" maxlength="80">'
												+ '</li></ul>');
								break;
							}
						});

		// 질문 복제
		$container.on('click', '.duplicate-box', function(e) {
			var $original = $(this).closest('.re_wr_box');
			var $clone = $original.clone(false); // false로 이벤트 복제 방지

			// id, data-id, 선택항목들의 data-id 제거
			$clone.removeAttr('id');
			$clone.find('.choice_sel_active').removeAttr('data-id');
			$clone.find('.option-text').removeAttr('data-id');

			$original.after($clone);
			fn_initializeBox($clone);
		});

		// 다른 영역 클릭 시 선택 목록 숨기기
		$(document).on('click', function(e) {
			if (!$(e.target).closest('.choice_sel').length) {
				$('.choice_sel_list').hide();
			}
		});

		// 응답항목 영역을 sortable로 초기화하여 드래그 앤 드롭 기능 활성화
		$('.response_writing_area').sortable({
			handle : '.re_wr_box_move', // 드래그 핸들러로 사용할 요소
			placeholder : 'sort-placeholder', // 드래그 중 위치 표시자
			opacity : 0.6, // 드래그 중인 항목의 투명도
			cursor : 'move', // 드래그 시 커서 모양
			update : function(event, ui) {
				// 드래그 앤 드롭 완료 후 실행될 콜백
				console.log('항목 순서가 변경되었습니다.');
			}
		});

		// 드래그 핸들러 표시/숨김 처리
		$container.on({
			mouseenter : function() {
				$(this).find('.re_wr_box_move').show();
			},
			mouseleave : function() {
				$(this).find('.re_wr_box_move').hide();
			}
		}, '.re_wr_box');

		// 새 질문 박스 추가 함수
		function fn_addNewQuestionBox(type) {
			var template = '';
			// 객관식 질문

		// 박스 초기화 함수
		function fn_initializeBox($box) {
			// 체크박스인 경우 선택 가능 개수 드롭다운 초기화
			if ($box.find('.choice_sel_active').attr('data-type') === 'checkbox') {
				var $select = $box.find('.checkbox-limit');
				var optionCount = $box.find('.option-text').length;
				fn_generateCheckboxLimitOptions($select[0], optionCount);
			}

			// 단답형/장문형인 경우 글자수 제한 드롭다운 초기화
			if ([ 'subjective', 'long' ].includes($box.find(
					'.choice_sel_active').attr('data-type'))) {
				var $select = $box.find('.text-limit');
				fn_generateTextLimitOptions($select[0]);
			}
		}

		// 드롭다운 옵션 생성 함수
		function fn_generateTextLimitOptions(select) {
			const currentValue = $(select).val(); // 현재 선택된 값 저장
			select.innerHTML = ''; // 기존 옵션 초기화
			// 100부터 1500까지 100단위로 옵션 추가
			for (let i = 100; i <= 1500; i += 100) {
				const option = document.createElement('option');
				option.value = i;
				option.textContent = i + '자';

				// 이전 선택값과 같으면 selected 속성 추가
				if (currentValue == i) {
					option.selected = true;
				}
				select.appendChild(option);
			}
		}

		/// 체크박스 선택 개수 드롭다운 옵션 생성 함수
		function fn_generateCheckboxLimitOptions(select, totalOptions) {
			const currentValue = $(select).val(); // 현재 선택된 값 저장
			select.innerHTML = ''; // 기존 옵션 초기화
			for (let i = 1; i <= totalOptions; i++) {
				const option = document.createElement('option');
				option.value = i;
				option.textContent = i + '개';

				// 이전 선택값과 같으면 selected 속성 추가
				if (currentValue == i) {
					option.selected = true;
				}
				select.appendChild(option);
			}
		}

		// 응답항목 추가 버튼 클릭 이벤트
		$(document).off('click', '.app_btn').on('click', '.app_btn',
				function() {
					const $newBox = $(fn_addNewQuestionBox('subjective'));

					if ($newBox && $newBox.length) {
						const select = $newBox.find('.text-limit')[0];
						if (select && select.options.length === 0) {
							fn_generateTextLimitOptions(select);
						}

						// 새로 추가된 박스로 스크롤
						$('html, body').animate({
							scrollTop : $newBox.offset().top - 100
						}, 500);

						// 질문 입력 필드에 포커스
						$newBox.find('.question-title').focus();
					}

					return false;
				});

		// 기존 박스들 초기화
		function fn_initializeExistingBoxes() {
			$('.re_wr_box').each(function() {
				fn_initializeBox($(this));
			});
		}

		// 개인정보 항목 설정 체크 여부 확인 함수
		function isPersonalInfoEnabled(artclCd) {
			return $('#useYn_' + artclCd).is(':checked');
		}

		// 필수 체크박스 클릭 이벤트 핸들러
		$('[id^=esntlYn_]').click(function(e) {
			var artclCd = this.id.split('_')[1];
			if (!isPersonalInfoEnabled(artclCd)) {
				e.preventDefault(); // 체크 동작 방지
				alert('체크박스를 클릭하여 해당 개인정보 항목을 활성화 해주시기 바랍니다.');
				return false;
			}

		});

		// 설정(useYn) 체크박스가 해제될 때 필수 체크도 함께 해제
		$('[id^=useYn_]').change(function() {
			var artclCd = this.id.split('_')[1];
			if (!this.checked) {
				$('#esntlYn_' + artclCd).prop('checked', false);
			}
		});

		// textarea 글자수 제한
		$(document).on('input', '.question-title', function() {
			var maxLength = 2000;
			var currentLength = $(this).val().length;

			if (currentLength > maxLength) {
				// 2000자를 초과하는 입력 제한
				$(this).val($(this).val().substring(0, maxLength));
				alert('최대 2000자까지 입력 가능합니다.');
			}
		});
