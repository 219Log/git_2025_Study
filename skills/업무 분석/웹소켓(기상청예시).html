   ↓                          ↓

                          ↕
                     계속 주고받기 가능

2. 코드 계층별 구성 (한줄씩 주석 포함)

WebSocketConfig.java

@Configuration
@EnableWebSocket // WebSocket 기능 활성화
public class WebSocketConfig implements WebSocketConfigurer {

    @Autowired
    private WeatherSocketHandler handler; // 실제 메시지 전송 로직을 가진 핸들러

    @Override
    public void registerWebSocketHandlers(WebSocketHandlerRegistry registry) {
        // WebSocket 연결 경로 설정: ws://서버/ws/weather
        registry.addHandler(handler, "/ws/weather")
                .setAllowedOrigins("*"); // 모든 출처 허용 (CORS 우회)
    }
}

WeatherSocketHandler.java

@Component
public class WeatherSocketHandler extends TextWebSocketHandler {

    private List<WebSocketSession> sessions = new ArrayList<>();

    // 클라이언트가 WebSocket 연결하면 자동 실행됨
    @Override
    public void afterConnectionEstablished(WebSocketSession session) {
        sessions.add(session); // 연결된 클라이언트 저장
    }

    // 서비스에서 변경된 날씨 데이터를 push할 때 호출
    public void broadcastUpdate(List<WeatherDto> data) {
        try {
            String json = new Gson().toJson(data); // 객체 → JSON 문자열
            TextMessage msg = new TextMessage(json);

            for (WebSocketSession s : sessions) {
                if (s.isOpen()) {
                    s.sendMessage(msg); // 메시지 전송
                }
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

weather.jsp(클라이언트 화면)

<table>
  <thead>
    <tr><th>날짜</th><th>기온</th><th>하늘 상태</th></tr>
  </thead>
  <tbody id="weatherTableBody">
    <!-- 실시간 데이터 들어올 자리 -->
  </tbody>
</table>

<script>
  const socket = new WebSocket("ws://localhost:8080/ws/weather");

  socket.onmessage = function(event) {
    const data = JSON.parse(event.data); // 서버가 보낸 JSON을 객체로 파싱
    const tbody = document.getElementById("weatherTableBody");
    tbody.innerHTML = ""; // 기존 테이블 비우기

    data.forEach(item => {
      const row = `
        <tr>
          <td>${item.fcstDate}</td>
          <td>${item.fcstValue}℃</td>
          <td>${item.sky}</td>
        </tr>`;
      tbody.innerHTML += row;
    });
  };
</script>
