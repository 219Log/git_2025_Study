클로저란?
클로저는 함수가 자신이 생성된 환경(변수들)을 기억하고 접근할 수 있는 JavaScript의 핵심 메커니즘입니다:

// 클로저 예시 1: 이벤트 핸들러에서 외부 변수 사용
$(document).on('click', '.app_btn', function() {
    // 외부에서 선언된 fn_addNewQuestionBox 함수 접근
    const $newBox = $(fn_addNewQuestionBox('subjective')); 
    // 함수 종료 후에도 $newBox 변수 유지 및 사용
    $newBox.find('.question-title').focus();
});

// 클로저 예시 2: 각 반복에서 생성된 환경 유지
$('.re_wr_box').each(function() {
    var $box = $(this);  // 각 반복마다 다른 $box
    
    // 이 이벤트 핸들러는 자신의 $box를 "기억"
    $box.find('.delete-box').click(function() {
        $box.remove();  // 클릭 시 자신의 부모 박스 제거
    });
});
